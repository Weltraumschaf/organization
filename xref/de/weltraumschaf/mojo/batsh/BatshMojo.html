236     private List<String> getCommandArguments() {
237         List<String> args = new ArrayList<String>();
238 
239         if (getOutputDirectory() != null) {
240             args.add("-o");
241             args.add(outputDirectory.getAbsolutePath());
242         }
243 
244         // Where do we want ANTLR to look for .tokens and import grammars?
245         if (getLibDirectory() != null && getLibDirectory().isDirectory()) {
246             args.add("-lib");
247             args.add(libDirectory.getAbsolutePath());
248         }
249 
250         // Next we need to set the options given to us in the pom into the
251         // tool instance we have created.
252         if (atn) {
253             args.add("-atn");
254         }
255 
256         if (encoding != null && !encoding.isEmpty()) {
257             args.add("-encoding");
258             args.add(encoding);
259         }
260 
261         if (listener) {
262             args.add("-listener");
263         } else {
264             args.add("-no-listener");
265         }
266 
267         if (visitor) {
268             args.add("-visitor");
269         } else {
270             args.add("-no-visitor");
271         }
272 
273         if (treatWarningsAsErrors) {
274             args.add("-Werror");
275         }
276 
277         if (forceATN) {
278             args.add("-Xforce-atn");
279         }
280 
281         if (options != null) {
282             for (Map.Entry<String, String> option : options.entrySet()) {
283                 args.add(String.format("-D%s=%s", option.getKey(), option.getValue()));
284             }
285         }
286 
287         if (arguments != null) {
288             args.addAll(arguments);
289         }
290 
291         return args;
292     }

294     public Set<String> getIncludesPatterns() {
295         if (includes == null || includes.isEmpty()) {
296             return Collections.singleton("**/*.g4");
297         }
298         return includes;

301     /**
302      * Given the source directory File object and the full PATH to a grammar, produce the path to the named grammar file
303      * in relative terms to the {@code sourceDirectory}. This will then allow ANTLR to produce output relative to the
304      * base of the output directory and reflect the input organization of the grammar files.
305      *
306      * @param sourceDirectory The source directory {@link File} object
307      * @param grammarFileName The full path to the input grammar file
308      * @return The path to the grammar file relative to the source directory
309      */
310     private String findSourceSubdir(File sourceDirectory, String grammarFileName) {
311         String srcPath = sourceDirectory.getPath() + File.separator;
312 
313         if (!grammarFileName.startsWith(srcPath)) {
314             throw new IllegalArgumentException("expected " + grammarFileName + " to be prefixed with " + sourceDirectory);
315         }
316 
317         File unprefixedGrammarFileName = new File(grammarFileName.substring(srcPath.length()));
318         if (unprefixedGrammarFileName.getParent() == null) {
319             return "";
320         }
321 
322         return unprefixedGrammarFileName.getParent() + File.separator;
323     }
